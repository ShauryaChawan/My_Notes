<h1 style="text-align: center">5. Java Loops II</h1>

```java
import java.util.*;

public class Solution {

  public static void main(String[] args) {
    Scanner sc = new Scanner(System.in);
    int q = sc.nextInt();
    sc.nextLine();
    while (q != 0) {
      int a = sc.nextInt();
      int b = sc.nextInt();
      int n = sc.nextInt();

      for (int i = 0; i < n; i++) {
        a += (((int) Math.pow(2, i)) * b);
        System.out.print(a + " ");
      }
      System.out.println();
      q--;
    }
    sc.close();
  }
}
```
