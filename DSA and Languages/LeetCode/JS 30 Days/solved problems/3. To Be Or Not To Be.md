# 2704. To Be Or Not To Be

```js
var expect = function (val) {
  return {
    toBe: function (otherValue) {
      if (val === otherValue) {
        return true;
      } else {
        throw new Error("Not Equal");
      }
    },
    notToBe: function (otherValue) {
      if (val !== otherValue) {
        return true;
      } else {
        throw new Error("Equal");
      }
    },
  };
};
```
